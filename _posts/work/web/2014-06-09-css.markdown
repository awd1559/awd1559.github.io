# sass

#### 定义变量
$fontStack: Helvetica, sans-serif;
$primaryColor: #333;

body{
	font-family: $fontStack;
	color: $primaryColor;
}


#### 嵌套
nav{
	ul{	}
	li{	}
}

==> nav ul{} 和 nav li {}

#### 导入
@import ‘reset’;	导入_reset.scss



#### mixin
@mixin box-sizing ($sizing) { 
  -webkit-box-sizing:$sizing; 
  -moz-box-sizing:$sizing; 
  box-sizing:$sizing; 
} 
.box-border { 
  border:1px solid #ccc; 
  @include box-sizing(border-box); 
}


#### 扩展
.message{
  border:1px solid #ccc;
}
.sucess{
  @extend .message;
  border-color:green;
}
